#### 邮件脚本mail.py(注意，python2.6/2.7版本没问题，python3有问题)
```
#!/usr/bin/python
#coding:utf-8
import smtplib
from email.mime.text import MIMEText
import sys
mail_host = 'smtp.163.com'
mail_user = 'test@163.com'
mail_pass = 'your_mail_password'
mail_postfix = '163.com'
def send_mail(to_list,subject,content):
    me = "zabbix 监控告警平台"+"<"+mail_user+"@"+mail_postfix+">"
    msg = MIMEText(content, 'plain', 'utf-8')
    msg['Subject'] = subject
    msg['From'] = me
    msg['to'] = to_list
    try:
        s = smtplib.SMTP()
        s.connect(mail_host)
        s.login(mail_user,mail_pass)
        s.sendmail(me,to_list,msg.as_string())
        s.close()
        return True
    except Exception,e:
        print str(e)
        return False
if __name__ == "__main__":
    send_mail(sys.argv[1], sys.argv[2], sys.argv[3])
```

#!/bin/bash
##该脚本用来监控网站的502问题
##作者：阿铭
##日期：2018-09-16
##版本：v0.1

t=`date -d "-1 min" +"%Y:%H:%M:[0-5][0-9]"`
log="/data/logs/access.log"
#假设mail.py已经写好，并放在/usr/local/sbin/下
mail_script="/usr/local/sbin/mail.py"
mail_user=aming@aminglinux.com

n=`grep $t $log|grep -n " 502 "`
if [ $n -gt 50 ]
then
    python $mail_script $mail_user "网站有502" "1分钟内出现了$n次"
fi
